var numero_aleatorio = Int.random(in: 1...100)
var intentos = 0
 
 
print("¡Bienvenido!")
print("He elegido un número entre 1 y 100. ¿Crees que puedes adivinarlo?")

func validar_entrada_a_numero() -> Int {
    var entrada_del_usuario: String? = nil
    var numero_valido: Int? = nil

    while true {
        print("Por favor introduce un número")
        entrada_del_usuario = readLine()
        
        if entrada_del_usuario == nil {
            print("Que paso mi hacker, creías que había una vulnerabilidad?")
        }
        
        numero_valido = Int(entrada_del_usuario!) 


        if numero_valido != nil {
            if numero_valido! == numero_aleatorio {
                print("¡Felicidades! Adivinaste el número en solo \(intentos + 1) intentos.")
                return numero_valido!
            } else {
                if numero_valido! < numero_aleatorio {
                    print("El número correcto es mayor.")
                } else {
                    print("El número correcto es menor.")
                }
                intentos += 1
            }
        } else {
            print("Lo que introdujiste no fue un número, por favor usa números y no letras")
        }
    }
}

var respuesta_del_usuario = validar_entrada_a_numero()
